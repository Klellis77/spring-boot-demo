1- Java installation
  - java -version
2- Maven installation [Gradle can be used]
  - mvn -version
3- Clone Spring based web application
  - git clone https://github.com/spring-guides/gs-spring-boot-docker.git
4- Change directory
  - cd gs-spring-boot-docker/complete
5- Build jar
  - ./mvnw package
6- Run application
  - java -jar target/spring-boot-docker-complete-0.0.1-SNAPSHOT.jar
7- Create a Docker file
    FROM openjdk:8-jdk-alpine
    COPY target/spring-boot-docker-complete-0.0.1-SNAPSHOT.jar app.jar
    ENTRYPOINT ["java","-jar","/app.jar"]
8- Build docker image
  - docker build -t springio/gs-spring-boot-docker .
9- Run docker image
  - docker run -p 8080:8080 springio/gs-spring-boot-docker
